cmake_minimum_required(VERSION 3.12)
project(PingTracer)

# Set C++ standard
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Set architecture
set(ARCHITECTURE "x64")

# Define output directories
set(OUTPUT_DIR "${CMAKE_SOURCE_DIR}/bin/${CMAKE_BUILD_TYPE}-${CMAKE_SYSTEM_NAME}-${ARCHITECTURE}/${PROJECT_NAME}")

# Set output directories for different types of output
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${OUTPUT_DIR})
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${OUTPUT_DIR})
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${OUTPUT_DIR})

# Define include directories
set(INCLUDE_DIR ${CMAKE_SOURCE_DIR}/PingTracer)
set(SDL_INCLUDE_DIR ${INCLUDE_DIR}/vendor/SDL/include)
set(IMGUI_INCLUDE_DIR ${INCLUDE_DIR}/vendor/ImGui)
set(IMPLOT_INCLUDE_DIR ${INCLUDE_DIR}/vendor/ImPlot)
set(ICMPLIB_INCLUDE_DIR ${INCLUDE_DIR}/vendor/icmplib)
set(PLF_NANOTIMER_INCLUDE_DIR ${INCLUDE_DIR}/vendor/plf_nanotimer)
set(STB_INCLUDE_DIR ${INCLUDE_DIR}/vendor/stb)

add_subdirectory(${CMAKE_SOURCE_DIR}/PingTracer/vendor/SDL)

# Add executable
add_executable(PingTracer WIN32)

# Add source files
target_sources(PingTracer PRIVATE
    ${CMAKE_SOURCE_DIR}/PingTracer/src/Source.cpp
    ${CMAKE_SOURCE_DIR}/PingTracer/src/App/App.h
    ${CMAKE_SOURCE_DIR}/PingTracer/src/App/App.cpp
    ${CMAKE_SOURCE_DIR}/PingTracer/src/App/AppColours.h
    ${CMAKE_SOURCE_DIR}/PingTracer/src/App/AppColours.cpp
    ${CMAKE_SOURCE_DIR}/PingTracer/src/App/AppIcon.h
    ${CMAKE_SOURCE_DIR}/PingTracer/src/App/AppIcon.cpp
    ${CMAKE_SOURCE_DIR}/PingTracer/src/App/AppSettings.h
    ${CMAKE_SOURCE_DIR}/PingTracer/src/App/AppSettings.cpp
    ${CMAKE_SOURCE_DIR}/PingTracer/src/PingTracer/MultithreadingWorker.cpp
    ${CMAKE_SOURCE_DIR}/PingTracer/src/PingTracer/MultithreadingWorker.h
    ${CMAKE_SOURCE_DIR}/PingTracer/src/PingTracer/PingTracer.cpp
    ${CMAKE_SOURCE_DIR}/PingTracer/src/PingTracer/PingTracer.h
    ${CMAKE_SOURCE_DIR}/PingTracer/src/PingTracer/PingTracerSettings.h
    ${CMAKE_SOURCE_DIR}/PingTracer/vendor/imgui/imconfig.h
    ${CMAKE_SOURCE_DIR}/PingTracer/vendor/imgui/imgui.h
    ${CMAKE_SOURCE_DIR}/PingTracer/vendor/imgui/imgui.cpp
    ${CMAKE_SOURCE_DIR}/PingTracer/vendor/imgui/imgui_demo.cpp
    ${CMAKE_SOURCE_DIR}/PingTracer/vendor/imgui/imgui_draw.cpp
    ${CMAKE_SOURCE_DIR}/PingTracer/vendor/imgui/imgui_internal.h
    ${CMAKE_SOURCE_DIR}/PingTracer/vendor/imgui/imgui_tables.cpp
    ${CMAKE_SOURCE_DIR}/PingTracer/vendor/imgui/imgui_widgets.cpp
    ${CMAKE_SOURCE_DIR}/PingTracer/vendor/imgui/imstb_rectpack.h
    ${CMAKE_SOURCE_DIR}/PingTracer/vendor/imgui/imstb_textedit.h
    ${CMAKE_SOURCE_DIR}/PingTracer/vendor/imgui/imstb_truetype.h
    ${CMAKE_SOURCE_DIR}/PingTracer/vendor/imgui/backends/imgui_impl_sdlrenderer3.cpp
    ${CMAKE_SOURCE_DIR}/PingTracer/vendor/imgui/backends/imgui_impl_sdlrenderer3.h
    ${CMAKE_SOURCE_DIR}/PingTracer/vendor/imgui/backends/imgui_impl_sdl3.cpp
    ${CMAKE_SOURCE_DIR}/PingTracer/vendor/imgui/backends/imgui_impl_sdl3.h
    ${CMAKE_SOURCE_DIR}/PingTracer/vendor/implot/implot.h
    ${CMAKE_SOURCE_DIR}/PingTracer/vendor/implot/implot.cpp
    ${CMAKE_SOURCE_DIR}/PingTracer/vendor/implot/implot_demo.cpp
    ${CMAKE_SOURCE_DIR}/PingTracer/vendor/implot/implot_internal.h
    ${CMAKE_SOURCE_DIR}/PingTracer/vendor/implot/implot_items.cpp
    ${CMAKE_SOURCE_DIR}/PingTracer/vendor/stb/stb_image.h
)

# Set include directories
target_include_directories(PingTracer PRIVATE
    ${INCLUDE_DIR}/src
    ${SDL_INCLUDE_DIR}
    ${IMGUI_INCLUDE_DIR}
    ${IMPLOT_INCLUDE_DIR}
    ${ICMPLIB_INCLUDE_DIR}
    ${PLF_NANOTIMER_INCLUDE_DIR}
    ${STB_INCLUDE_DIR}
)

# Link libraries
target_link_libraries(PingTracer PRIVATE SDL3-static)

# Platform-specific settings
if(WIN32)
    target_link_libraries(PingTracer PRIVATE setupapi winmm version)

    target_compile_definitions(PingTracer PRIVATE WINDOWS)
    
    # Add resource files
    target_sources(PingTracer PRIVATE ${CMAKE_SOURCE_DIR}/PingTracer/PingTracer.rc)
    target_sources(PingTracer PRIVATE
        ${CMAKE_SOURCE_DIR}/PingTracer/src/App/Icon.h
        ${CMAKE_SOURCE_DIR}/PingTracer/src/App/Icon.cpp
    )

    # Post-build commands
    add_custom_command(TARGET PingTracer POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy 
            ${CMAKE_SOURCE_DIR}/PingTracer/src/PingTracerConfig.ini
            $<TARGET_FILE_DIR:PingTracer>
        COMMAND ${CMAKE_COMMAND} -E copy 
            ${CMAKE_SOURCE_DIR}/PingTracer/Roboto.ttf
            $<TARGET_FILE_DIR:PingTracer>
    )
elseif(APPLE)
    set_target_properties(PingTracer PROPERTIES
        MACOSX_BUNDLE TRUE
    )
elseif(UNIX)
    # Linux-specific settings if needed
endif()

# Configuration-specific settings
set(CMAKE_CONFIGURATION_TYPES Debug Release)

# Set startup project
set_property(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY VS_STARTUP_PROJECT PingTracer)

# Set DPI awareness
if(WIN32)
    set_property(TARGET PingTracer PROPERTY VS_DPI_AWARE "PerMonitor")
endif()
